49/*
50 * This file contains very specialized methods for updating the context
51 * class loader of a thread, because this is where they can be implemented
52 * without the overhead of several calls to wrappers defined here.
53 *
54 * More lightweight implementations of those can be chosen if the selected
55 * thread policy precludes native access from any but the primordial thread.
56 */

69void pljava_JNI_threadInitialize(bool manageLoader)
70{
71 if ( ! manageLoader )
72 {
73 JNI_loaderUpdater = _noopUpdater;
74 JNI_loaderRestorer = _noopRestorer;
75 return;
76 }
77
78 s_Thread_class = JNI_newGlobalRef(PgObject_getJavaClass(
79 "java/lang/Thread"));
80 s_Thread_currentThread = PgObject_getStaticJavaMethod(
81 s_Thread_class,
82 "currentThread",
83 "()"
84 "Ljava/lang/Thread;");
85 s_Thread_contextLoader = JNI_getFieldIDOrNull(s_Thread_class,
86 "contextClassLoader", "Ljava/lang/ClassLoader;");
87
88 if ( NULL == s_Thread_contextLoader )
89 {
90 ereport(WARNING, (
91 errmsg("unable to manage thread context classloaders in this JVM")
92 ));
93 JNI_loaderUpdater = _noopUpdater;
94 JNI_loaderRestorer = _noopRestorer;
95 }
96 else if ( s_refuseOtherThreads || ! s_doMonitorOps )
97 {
98 s_threadObject =
99 JNI_newGlobalRef(
100 JNI_callStaticObjectMethod(
101 s_Thread_class, s_Thread_currentThread));
102 JNI_loaderUpdater = _lightUpdater;
103 JNI_loaderRestorer = _lightRestorer;
104 }
105 else
106 {
107 JNI_loaderUpdater = _heavyUpdater;
108 JNI_loaderRestorer = _heavyRestorer;
109 }
110}

113/*
114 * BEGIN_JAVA and END_JAVA are used in JNI wrappers that are not expected to
115 * invoke Java methods; all they do is play the game with the scope of the JNI
116 * env value that was devised to fail fast if the intended pattern for PL/Java's
117 * JNI usage isn't followed.
118 *
119 * BEGIN_CALL and END_CALL add to that the releasing of the "THREADLOCK"
120 * monitor when calling into Java, and reacquiring it on return, that support
121 * the java_thread_pg_entry=allow mode of operation, and also checking for
122 * exceptions and turning them into PostgreSQL ereports.
123 *
124 * The _MONITOR_HELD flavors of those skip the monitor operations but still do
125 * the exception checks. They are used in a select few *Locked flavors of
126 * method call wrappers used where only known and lightweight Java methods will
127 * be invoked and not arbitrary methods of user code.
128 */
129#define BEGIN_JAVA { JNIEnv* env = jniEnv; jniEnv = 0;
130#define END_JAVA jniEnv = env; }
131
132#define BEGIN_CALL \
133 BEGIN_JAVA \
134 if(s_doMonitorOps && ((*env)->MonitorExit(env, s_threadLock) < 0)) \
135 elog(ERROR, "Java exit monitor failure");
136
137#define END_CALL endCall(env); }
138
139#define BEGIN_CALL_MONITOR_HELD \
140 BEGIN_JAVA
141
142#define END_CALL_MONITOR_HELD endCallMonitorHeld(env); }

144static void elogExceptionMessage(JNIEnv* env, jthrowable exh, int logLevel)
145{
146 StringInfoData buf;
147 int sqlState = ERRCODE_INTERNAL_ERROR;
148 JNIEnv* saveEnv = jniEnv;
149 jclass exhClass = (*env)->GetObjectClass(env, exh);
150 jstring jtmp =
151 (jstring)(*env)->CallObjectMethod(env, exhClass, Class_getName);
152 /* ExceptionOccurred check is found below ... */
153
154 initStringInfo(&buf);
155
156 jniEnv = env; /* Used by the String operations */
157
158 if ( 0 == (*env)->ExceptionOccurred(env) ) /* ... here */
159 String_appendJavaString(&buf, jtmp);
160 else
161 {
162 (*env)->ExceptionClear(env);
163 appendStringInfoString(&buf, "<unknown Java class>");
164 }
165
166 (*env)->DeleteLocalRef(env, exhClass);
167 (*env)->DeleteLocalRef(env, jtmp);
168
169 jtmp = (jstring)(*env)->CallObjectMethod(env, exh, Throwable_getMessage);
170 if ( 0 != (*env)->ExceptionOccurred(env) )
171 {
172 (*env)->ExceptionClear(env);
173 jtmp = 0;
174 }
175
176 if(jtmp != 0)
177 {
178 appendStringInfoString(&buf, ": ");
179 String_appendJavaString(&buf, jtmp);
180 (*env)->DeleteLocalRef(env, jtmp);
181 }
182
183 if((*env)->IsInstanceOf(env, exh, SQLException_class))
184 {
185 jtmp = (*env)->CallObjectMethod(env, exh, SQLException_getSQLState);
186 if ( 0 != (*env)->ExceptionOccurred(env) )
187 {
188 (*env)->ExceptionClear(env);
189 jtmp = 0;
190 }
191
192 if(jtmp != 0)
193 {
194 char* s = String_createNTS(jtmp);
195 (*env)->DeleteLocalRef(env, jtmp);
196
197 if(strlen(s) >= 5)
198 sqlState = MAKE_SQLSTATE(s[0], s[1], s[2], s[3], s[4]);
199 pfree(s);
200 }
201 }
202 jniEnv = saveEnv;
203 ereport(logLevel, (errcode(sqlState), errmsg("%s", buf.data)));
204}

206static void printStacktrace(JNIEnv* env, jobject exh)
207{
208#ifndef _MSC_VER
209 if(DEBUG1 >= log_min_messages || DEBUG1 >= client_min_messages)
210#else
211 /* This is gross, but only happens as often as an exception escapes Java
212 * code to be rethrown. There is some renewed interest on pgsql-hackers to
213 * find a good answer for the MSVC PGDLLIMPORT nonsense, and all of this
214 * handling of exceptions and logging could stand some rework anyway.
215 */
216 if ( 0 == strncmp("debug", PG_GETCONFIGOPTION("log_min_messages"), 5)
217 || 0 == strncmp("debug", PG_GETCONFIGOPTION("client_min_messages"), 5) )
218#endif
219 {
220 int currLevel = Backend_setJavaLogLevel(DEBUG1);
221 (*env)->CallVoidMethod(env, exh, Throwable_printStackTrace);
222 (*env)->ExceptionOccurred(env); /* sop for JNI exception-check check */
223 Backend_setJavaLogLevel(currLevel);
224 }
225}

227static void endCall(JNIEnv* env)
228{
229 jobject exh = (*env)->ExceptionOccurred(env);
230 if(exh != 0)
231 (*env)->ExceptionClear(env);
232
233 if(s_doMonitorOps && ((*env)->MonitorEnter(env, s_threadLock) < 0))
234 elog(ERROR, "Java enter monitor failure");
235
236 jniEnv = env;
237 if(exh != 0)
238 {
239 printStacktrace(env, exh);
240 if((*env)->IsInstanceOf(env, exh, ServerException_class))
241 {
242 /* Rethrow the server error.
243 */
244 jobject jed = (*env)->CallObjectMethod(env, exh, ServerException_getErrorData);
245 if ( 0 != (*env)->ExceptionOccurred(env) )
246 {
247 (*env)->ExceptionClear(env);
248 jed = 0;
249 }
250
251 if(jed != 0)
252 ReThrowError(pljava_ErrorData_getErrorData(jed));
253 }
254 /* There's no return from this call.
255 */
256 elogExceptionMessage(env, exh, ERROR);
257 }
258}

290bool beginNativeNoErrCheck(JNIEnv* env)
291{
292 if ( s_refuseOtherThreads && env != primordialJNIEnv )
293 {
294 env = JNI_setEnv(env);
295 Exception_throw(ERRCODE_INTERNAL_ERROR,
296 "Attempt by non-initial thread to enter PostgreSQL from Java");
297 JNI_setEnv(env);
298 return false;
299 }
300
301 if((env = JNI_setEnv(env)) != 0)
302 {
303 /* The backend is *not* awaiting the return of a call to the JVM
304 * so there's no way the JVM can be allowed to call out at this
305 * point.
306 */
307 Exception_throw(ERRCODE_INTERNAL_ERROR,
308 "An attempt was made to call a PostgreSQL backend function while main thread was not in the JVM");
309 JNI_setEnv(env);
310 return false;
311 }
312 return true;
313}

315bool beginNative(JNIEnv* env)
316{
317 if (!currentInvocation)
318 {
319 env = JNI_setEnv(env);
320 Exception_throw(ERRCODE_INTERNAL_ERROR, "An attempt was made to call a PostgreSQL backend function in a transaction callback. At the end of a transaction you may not access the database any longer.");
321 JNI_setEnv(env);
322 return false;
323 }
324
325 if(currentInvocation->errorOccurred)
326 {
327 /* An elog with level higher than ERROR was issued. The transaction
328 * state is unknown. There's no way the JVM is allowed to enter the
329 * backend at this point.
330 */
331 env = JNI_setEnv(env);
332 Exception_throw(ERRCODE_INTERNAL_ERROR,
333 "An attempt was made to call a PostgreSQL backend function after an elog(ERROR) had been issued");
334 JNI_setEnv(env);
335 return false;
336 }
337 return beginNativeNoErrCheck(env);
338}

1633/*
1634 * Implementations of the context class loader updater and restorer.
1635 * The loader reference passed in is not to be deleted. If saved anywhere,
1636 * a new global ref is to be taken, and later deleted when restored.
1637 */
1638
1639static inline void _updaterCommon(JNIEnv *env, jobject thread, jobject loader)
1640{
1641 jobject old = (*env)->GetObjectField(env, thread, s_Thread_contextLoader);
1642
1643 /*
1644 * If it is not already the loader we want, change it, and set
1645 * currentInvocation->savedLoader to restore it later. If this is
1646 * a top-level invocation, we don't care what it gets restored to, so lie,
1647 * and save loader there instead of old. If there are many consecutive
1648 * top-level calls with the same context loader, that will save work later.
1649 *
1650 * If it is already the loader we want, again we check for a top-level call,
1651 * and can leave currentInvocation->savedLoader completely unset in that
1652 * case, so the restore call will be skipped completely. If not a top-level
1653 * call, though, see that it gets restored to what the caller might expect,
1654 * even if it somehow got changed.
1655 */
1656
1657 if ( ! (*env)->IsSameObject(env, old, loader) )
1658 {
1659 (*env)->SetObjectField(env, thread, s_Thread_contextLoader, loader);
1660
1661 currentInvocation->savedLoader = (*env)->NewGlobalRef(env,
1662 ( NULL == currentInvocation->previous ) ? loader : old);
1663 }
1664 else if ( NULL != currentInvocation->previous )
1665 currentInvocation->savedLoader = (*env)->NewGlobalRef(env, old);
1666
1667 (*env)->DeleteLocalRef(env, old);
1668}
